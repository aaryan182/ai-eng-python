Pydantic Models

Pydantic is: a validator, a type- character , a serializer , the base of LLM tool definitions

from pydantic import BaseModel

class SummarizeInput(BaseModel):
    text: str

@app.post("/summarize")
def summarize(data: SummarizeInput):
    return {"summary": data.text[:20] + "..."}
